//El programa recibe n e imprime los n primeros numeros primos

def void imprimir(int a){
	out(a);
}

def bool esPrimo(int n){
	bool primo = true;
	int i = 2;
	while(i < (n/2) && primo){
		if(n%i == 0){
			primo = false;
		}
		i = i +1;
	}
	return primo;
}

	int contador = 1;
	int n = 0;
	int encontrados = 0;
	in(n);

	while(encontrados < n){
		if(esPrimo(contador)){
			imprimir(contador);
			encontrados = encontrados + 1;
		}

	}
}
